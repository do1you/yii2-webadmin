HTML调用

$access = ($this->module !== null ? $this->module->id . '@' : "") . $this->id .':'. $this->action->id;
$title = SysLdItem::dd('shop_authority',$access,true);
$title && $this->setPageTitle($title); // 写入标题

if($title!==null){ // 判断用户 是否有权限
	$result = $this->checkAuth();
	if($result===false){
		$this->error('您没有权限执行['.$title.']');
	}
}else{ // 未定义权限的页面一律放行
	$result = true;
}
